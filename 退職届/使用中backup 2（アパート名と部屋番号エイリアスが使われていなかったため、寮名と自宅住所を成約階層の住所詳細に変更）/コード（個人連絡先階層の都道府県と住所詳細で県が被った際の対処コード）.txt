											<xsl:variable name="pref" select="normalize-space(Sales.P_Candidate/Candidate/Person.U_6E906CC3D68B8B4F24A2BA8C5398E7)"/>
											<xsl:variable name="addr" select="normalize-space(Sales.U_A6F2AF68362CFCBC3833090D205198)"/>

											<xsl:choose>
											<!-- pref が空でない AND addr の先頭が pref と一致する場合 -->
											<xsl:when test="string-length($pref) &gt; 0 and substring($addr, 1, string-length($pref)) = $pref">
												<!-- pref を1回表示し、addr の先頭の pref 部分を切り落として続ける -->
												<xsl:value-of select="$pref"/>
												<xsl:value-of select="substring($addr, string-length($pref) + 1)"/>
											</xsl:when>
											<xsl:otherwise>
												<!-- 重複がなければ通常通り連結 -->
												<xsl:value-of select="concat($pref, $addr)"/>
											</xsl:otherwise>
											</xsl:choose>
